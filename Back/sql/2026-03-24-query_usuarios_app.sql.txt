--------- QUERYS PARA PEGAR OS VENDEDORES E GERENTES PARA O INSERT -------

---- GERENTES ----

select 
CPF_CNPJ_SEM_PONTOS,
EMPRESA_ID,
FUNCIONARIO_ID,
FUNCIONARIO_ID as vendedor_id,
NOME
from fato_funcionarios_acessos
WHERE GERENTE = 'S'
AND CPF_CNPJ_SEM_PONTOS IS NOT NULL
;


---- VENDEDORES ----
select
distinct func.CPF_CNPJ_SEM_PONTOS as login,
ven.sk_vendedor,
ven.vendedor_id,
func.funcionario_id,
ven.nome_vendedor,
ven.empresa_id

from dim_vendedor ven
left join fato_funcionarios_acessos func
on ven.vendedor_id = func.funcionario_id
where ven.sk_vendedor IS NOT NULL
    and ven.sk_vendedor <> 0
    and ven.sk_vendedor <> -1
    and CPF_CNPJ_SEM_PONTOS is not null
;
